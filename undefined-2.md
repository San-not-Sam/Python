```python
# Input 

def add(a, b):
  return a + b

def sub(a, b):
  return a - b

def mul(a, b):
  return a * b

def div(a, b):
  if b == 0:
    print("can't be divided by zero.")
    return
  return a / b

n1, n2 = 2, 0
print(add(n1, n2)) 
print(sub(n1, n2))
print(mul(n1, n2))
print(div(n1, n2))
```
